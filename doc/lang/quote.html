$.quote(str)

描述：

为字符串两边添加上双引号，主要用于JSON或模板等(不能使用JSON.stringify,它会清掉里面的函数)。

str
必需。String。
例子 
$.require("lang", function() {
    $.log($.quote("aaa\veee"), true);
    $.log("aaa\veee", true);
});

//版本1 只能处理单行
function quote(s) {
    (s += '').replace(/\\/g, '\\\\');
    var use = +(s.match(quote.q[1][0]) === null);
    return quote.q[0][use] + s.replace(quote.q[1][1 - use], '\\$1') + quote.q[0][use];
}
quote.q = [
    ['"', "'"],
    [/(')/g, /(")/g]
];
//版本2 太长
var quote = String.quote || (function() {
    var meta = {
        '\b': '\\b',
        '\t': '\\t',
        '\n': '\\n',
        '\f': '\\f',
        '\r': '\\r',
        '"': '\\"',
        '\\': '\\\\'
    },
    reg = /[\x00-\x1F\'\"\\\u007F-\uFFFF]/g,
        regFn = function(c) {
            if (c in meta) {
                return '\\' + meta[c];
            }
            var ord = c.charCodeAt(0);
            return ord < 0x20 ? '\\x0' + ord.toString(16) : ord < 0x7F ? '\\' + c : ord < 0x100 ? '\\x' + ord.toString(16) : ord < 0x1000 ? '\\u0' + ord.toString(16) : '\\u' + ord.toString(16)
        };
    return function(str) {
        return '"' + (str || "").replace(reg, regFn) + '"';
    }
})();
