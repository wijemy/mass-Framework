$.isFunction

描述：

判定是否是函数。

例子 
$.require("lang", function() {
    $.log($.isFunction($.noop), true);
    $.log($.isFunction(/test/i), true);
    $.log($.isFunction(document.childNodes), true);
    $.log($.isFunction(document.createElement("object")), true);
    $.log($.isFunction(document.createElement("embed")), true);
});
